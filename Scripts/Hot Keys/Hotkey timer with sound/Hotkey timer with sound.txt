var timerName = 'SimpleTimer'; // change for different hotkey + timerScript

// Delete old timer (if exists)
sp.DeleteTimer(timerName);

// from file
var timerScript = "sp.PlaySound('C:\\\\Windows\\\\Media\\\\Alarm10.wav');" +

// system beep sound
//var timerScript = "clr.System.Console.Beep(800, 100);" +

                  "sp.DeleteTimer('" + timerName + "');";

// Start new timer
sp.CreateTimer(timerName, timerDurationSec * 1000, 1, timerScript);
