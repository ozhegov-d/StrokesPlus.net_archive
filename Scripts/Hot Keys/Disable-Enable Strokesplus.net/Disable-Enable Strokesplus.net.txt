var wasOn = StrokesPlus.Configuration.Settings.CaptureEnabled;
if (wasOn) {
    StrokesPlus.Program.Disable();
    sp.Sleep(50); 
    showBottom("DISABLED", "255,0,0");
} else {
    StrokesPlus.Program.Enable();
    sp.Sleep(50); 
    showBottom("ENABLED", "0,255,0");
}

function showBottom(text, rgb){
    var info = new DisplayTextInfo();
    info.Title            = "";
    info.Message          = text;
    info.Duration         = 1000;
    info.Opacity          = 0.92;
    info.Location         = "bottom";
    info.MessageAlignment = "Center";
    info.TitleAlignment   = "Center";
    info.BackColor        = "0,0,0";
    info.ForeColor        = rgb;
    info.Padding          = 16;
    try {
        info.MessageFont = new System.Drawing.Font("Segoe UI", 20, System.Drawing.FontStyle.Bold);
    } catch(e) {}
    sp.DisplayText(info);
}
