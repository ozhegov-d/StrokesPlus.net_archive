  function parseDictVariants(json, pair) {
    var out = [];
    if (!json) return out;
    function addEntry(e) {
      if (!e || !e.tr) return;
      for (var k = 0; k < e.tr.length && out.length < dictMaxVariants; k++) {
        var t = e.tr[k];
        if (!t || !t.text) continue;
        var pos = "";
        if (t.pos) {
          var raw = (typeof t.pos === "string") ? t.pos : (t.pos.text || t.pos.code || "");
          pos = posFullName(raw);
        }
        var backs = [];
        if (t.mean && t.mean.length) {
          for (var m = 0; m < Math.min(5, t.mean.length); m++) {
            if (t.mean[m] && t.mean[m].text && backs.indexOf(t.mean[m].text) === -1) backs.push(t.mean[m].text);
          }
        }
        if (!backs.length && t.syn && t.syn.length) {
          for (var s = 0; s < Math.min(3, t.syn.length); s++) {
            if (t.syn[s] && t.syn[s].text && backs.indexOf(t.syn[s].text) === -1) backs.push(t.syn[s].text);
          }
        }
        out.push({ text: t.text, pos: pos, backs: backs });
      }
    }
    function walk(node, depth) {
      if (!node || depth > 6 || out.length >= dictMaxVariants) return;
      if (node.text && node.tr) { addEntry(node); return; }
      var keys = ['regular', 'synthetic'];
      for (var i = 0; i < keys.length; i++) {
        var arr = node[keys[i]];
        if (arr && arr.length) {
          for (var j = 0; j < arr.length && out.length < dictMaxVariants; j++) addEntry(arr[j]);
        }
      }
      for (var key in node) {
        if (Object.prototype.hasOwnProperty.call(node, key) && keys.indexOf(key) === -1) walk(node[key], depth + 1);
      }
    }
    if (json[pair]) walk(json[pair], 0); else walk(json, 0);
    return out;
  }

  function parseDictExamples(json, pair) {
    var out = [];
    if (!json || !json.result || !json.result.length) return out;

      if (s) out.push({ s: s, d: d });
    }
    return out;
  }

  function wrapFast(txt, len) {
    if (!txt || txt.length <= len) return txt.trim();
    var res = [], i = 0, j, k;
    while (i < txt.length) {
      j = i + len;
      if (j >= txt.length) { res.push(txt.substring(i).trim()); break; }
      k = txt.lastIndexOf(" ", j);
      if (k <= i) k = j;
      res.push(txt.substring(i, k).trim());
      i = k + 1;
    }
    return res.join("\n");
  }

      sections.push({ text: header, color: "white" });
      plainParts.push(header);

      if (dictError) {
        sections.push({ text: "Dictionary: unavailable", color: "gray" });
        plainParts.push("Dictionary: unavailable");
      } else if (variants.length) {
        sections.push({ text: "Dictionary:", color: "gray" });

          var v = variants[i];

        }
        var dictText = dictLines.join("\n");
        sections.push({ text: dictText, color: "white" });
        plainParts.push("Dictionary:\n" + dictText);
      }

        }
        var exText = exLines.join("\n");

        plainParts.push("Examples:\n" + exText);
      }
    } else {
      var transText = wrapFast(trText, wl);
      sections.push({ text: transText, color: "white" });
      plainParts.push(transText);
    }

    return { text: plainParts.join("\n\n"), sections: sections, wrapLen: wl };
  }

        try{f.Controls.Remove(lblSep);}catch(_){}
      }
      contentW = singleW;
      if (copyPosition==='up' && btnH) {
        positionButtonsRow(pad, pad);
        positionContentLabels(pad, pad+btnH+btnGap, cSize.sizes);
        contentH = pad+btnH+btnGap+cSize.height+pad;
      } else if (copyPosition==='up') {
        positionContentLabels(pad, pad, cSize.sizes);
        contentH = pad+cSize.height+pad;
      } else if (btnH) {
        positionContentLabels(pad, pad, cSize.sizes);
        positionButtonsRow(pad+cSize.height+btnGap, pad);
        contentH = pad+cSize.height+btnGap+btnH+pad;
      } else {
        positionContentLabels(pad, pad, cSize.sizes);
        contentH = pad+cSize.height+pad;
      }
    } else {
      createSplitLabels();
      if(!f.Controls.Contains(lblSrc)) f.Controls.Add(lblSrc);
      if(!f.Controls.Contains(lblSep)) f.Controls.Add(lblSep);
      lblSrc.BringToFront(); lblSep.BringToFront();
      lblSrc.Visible=true; lblSep.Visible=true;

      var W = cSize.width;
      if (2*W+sepBlock < btnRowW) W = Math.ceil((btnRowW-sepBlock)/2);

      lblSrc.Text = src;
      lblSrc.AutoSize = true;
      lblSrc.MaximumSize = new Drawing.Size(W, 0);

      var colW = Math.max(W, lblSrc.PreferredSize.Width, cSize.width);
      var maxH = Math.max(lblSrc.PreferredSize.Height, cSize.height);
      contentW = pad+colW+sepBlock+colW+pad;
      btnX = pad + (contentW - singleW);
      if (btnX < pad) btnX = pad;

      var contentY;
      if (copyPosition==='up' && btnH) {
        positionButtonsRow(pad, btnX);
        contentY = pad+btnH+btnGap;
        contentH = pad+btnH+btnGap+maxH+pad;
      } else if (copyPosition==='up') {
        contentY = pad; contentH = pad+maxH+pad;
      } else if (btnH) {
        contentY = pad;
        positionButtonsRow(contentY+maxH+btnGap, btnX);
        contentH = pad+maxH+btnGap+btnH+pad;
      } else {
        contentY = pad; contentH = pad+maxH+pad;
      }
      lblSrc.Location = new Drawing.Point(pad, contentY);
      lblSep.Location = new Drawing.Point(pad+colW+gap, contentY);
      lblSep.Height = maxH;
      positionContentLabels(pad+colW+sepBlock, contentY, cSize.sizes);
    }

    var totalW=contentW, totalH=contentH;

    if (historyVisible && histEnabled) {
      var histArr = _histCache || [];
      var histW = Math.max(Math.min(contentW, 420), 160);
      var histTextX = contentW + sepBlock;
      var lineH = Forms.TextRenderer.MeasureText("Ag",uiFont,new Drawing.Size(histW,0),Forms.TextFormatFlags.SingleLine).Height+4;
      if (histArr.length > histLimit) histArr = histArr.slice(0, histLimit);
      var n = histArr.length ? histArr.length : 1;
      var histH = pad + n*lineH + (n-1)*3 + pad;
      totalH = Math.max(contentH, histH);
      totalW = contentW + sepBlock + histW;

      var vsep = new Forms.Label();
      vsep.AutoSize=false; vsep.Width=2; vsep.Height=totalH-pad*2;
      vsep.BackColor=Drawing.Color.FromArgb(80,80,80); vsep.Text="";
      vsep.Location=new Drawing.Point(contentW+gap, pad);
      f.Controls.Add(vsep); vsep.BringToFront();
      historyControls.push(vsep); attachDrag(vsep);

      var hy = pad;
      if (!histArr.length) {
        makeHistoryLine("(пусто)",null,hy,histTextX,histW,lineH,true);
        hy += lineH+3;
      } else {
        for (var i=0;i<histArr.length;i++) {
          var sval = histArr[i]&&histArr[i].s ? String(histArr[i].s) : "";
          var oneLine = sval.replace(/\s+/g," ").trim();
          if (!oneLine) oneLine = "(пусто)";
          makeHistoryLine((i+1)+". "+oneLine, histArr[i], hy, histTextX, histW, lineH, false);
          hy += lineH+3;
        }
      }
    }

    lastContentW = contentW;
    f.AutoSize = false;
    f.Size = new Drawing.Size(totalW, totalH);
  }

  function clampToScreen() {
    try {
      var screen = Forms.Screen.FromPoint(f.Location);
      var wa = screen.WorkingArea;
      var x=f.Location.X, y=f.Location.Y;
      if(x<wa.Left)x=wa.Left; if(y<wa.Top)y=wa.Top;
      if(x+f.Width>wa.Right)x=wa.Right-f.Width;
      if(y+f.Height>wa.Bottom)y=wa.Bottom-f.Height;
      if(x<wa.Left)x=wa.Left; if(y<wa.Top)y=wa.Top;
      f.Location = new Drawing.Point(x,y);
    } catch(_) {}
  }

  function toggleSplit() {
    var oldRight = f.Location.X + lastContentW;
    splitMode = !splitMode;
    layoutAll();
    var newRight = f.Location.X + lastContentW;
    var dx = oldRight - newRight;
    if (dx!==0) f.Location = new Drawing.Point(f.Location.X+dx, f.Location.Y);
    clampToScreen();
    try{f.TopMost=true;f.BringToFront();f.Activate();}catch(_){}
  }

  function toggleHistory() {
    if (!histEnabled) return;
    historyVisible = !historyVisible;
    layoutAll();
    clampToScreen();
    try{f.TopMost=true;f.BringToFront();f.Activate();}catch(_){}
  }

    var btn = new Forms.Button();

    btn.FlatStyle=Forms.FlatStyle.Flat;
    btn.BackColor=Drawing.Color.White;
    btn.ForeColor=Drawing.Color.Black;
    btn.Click.connect(onClick);
    f.Controls.Add(btn);
    return btn;
  }
