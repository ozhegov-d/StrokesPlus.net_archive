  function decodeJWTExp(token) {
    try {
      var parts = token.split('.');
      if (parts.length !== 3) return null;
      var base64 = parts[1].replace(/-/g, '+').replace(/_/g, '/');
      while (base64.length % 4) base64 += '=';
      var bytes = clr.System.Convert.FromBase64String(base64);
      var payloadJson = clr.System.Text.Encoding.UTF8.GetString(bytes);
      var payload = JSON.parse(payloadJson);
      if (payload.exp) return payload.exp * 1000;
    } catch(e) {}
    return null;
  }

  function getTranslateToken() {

          var remainingSec = Math.floor((__translateTokenExp - now) / 1000);
          if (remainingSec > 0) {
            log("Using cached token (~" + Math.floor(remainingSec / 60) + "m " + (remainingSec % 60) + "s left)", "Auth");
          } else {
            log("Using cached token (JWT exp " + Math.abs(Math.floor(remainingSec / 60)) + "m ago, may still work until API returns 401)", "Auth");
          }
        } else {
          log("Using cached token (exp unknown)", "Auth");
        }
      }
      return __translateToken;
    }
    try {
      if (DEBUG_LOGS) log("Requesting new token...", "Auth", Drawing.Color.Orange);
      var res = __translateHttpClient.GetAsync("https://edge.microsoft.com/translate/auth").Result;
      if (!res.IsSuccessStatusCode) { resetToken(); return null; }

      return token;

  function isTextInLanguage(text, lang) {
    switch(lang) {

      case 'zh': case 'zh-Hans': case 'zh-Hant': return /[\u4e00-\u9fff]/i.test(text);
      case 'ja': return /[\u3040-\u309f\u30a0-\u30ff\u4e00-\u9fff]/i.test(text);
      case 'ko': return /[\uac00-\ud7a3]/i.test(text);
      case 'ar': return /[\u0600-\u06ff]/i.test(text);
      default: return false;
    }
  }

  function formatDictionaryText(lookupJson, examplesJson, fallbackTranslation, sourceText, translateFrom, translateTo, dictFrom, dictTo) {
    function cleanSentence(s) {
      return String(s || "").replace(/\s+/g, " ").trim();
    }
    var lines = [];
    if (fallbackTranslation) {
      var tFrom = translateFrom || "auto";
      var tTo = translateTo || tl;
      var samePair = dictFrom && dictTo && (String(dictFrom) === String(tFrom)) && (String(dictTo) === String(tTo));
      var dictPair = (!samePair && dictFrom && dictTo) ? (", dict (" + dictFrom + " -> " + dictTo + ")") : "";
      lines.push(sourceText + " - " + fallbackTranslation + " (" + tFrom + " -> " + tTo + ")" + dictPair, "");
    }
    // Fallback when lookup has no translations: keep translate header line or plain tr only.
    if (!lookupJson || !lookupJson[0] || !lookupJson[0].translations || lookupJson[0].translations.length === 0) {
      if (lines.length) return lines.join("\n").trim();
      return fallbackTranslation || "";
    }
    var entry = lookupJson[0];
    lines.push("Dictionary:");
    var maxItems = Math.min(5, entry.translations.length);
    for (var i = 0; i < maxItems; i++) {
      var t = entry.translations[i];
      var pos = t.pos ? (" [" + t.pos + "]") : "";
      var conf = (typeof t.confidence === "number") ? (" (" + Math.round(t.confidence * 100) + "%)") : "";
      lines.push((i + 1) + ". " + t.displayTarget + pos + conf);
      if (t.backTranslations && t.backTranslations.length) {
        var backs = [];
        for (var j = 0; j < Math.min(3, t.backTranslations.length); j++) backs.push(t.backTranslations[j].displayText);
        lines.push("   <- " + backs.join(", "));
      }
    }

    if (examplesJson && examplesJson[0] && examplesJson[0].examples && examplesJson[0].examples.length) {
      lines.push("", "Examples:");
      for (var k = 0; k < Math.min(3, examplesJson[0].examples.length); k++) {
        var ex = examplesJson[0].examples[k];
        lines.push("- " + cleanSentence((ex.sourcePrefix || "") + (ex.sourceTerm || "") + (ex.sourceSuffix || "")));
        lines.push("  " + cleanSentence((ex.targetPrefix || "") + (ex.targetTerm || "") + (ex.targetSuffix || "")));
      }
    }
    return lines.join("\n");
  }

  function wrapFast(txt, len) {
    if (!txt || txt.length <= len) return txt.trim();
    var res = [], i = 0, j, k;
    while (i < txt.length) {
      j = i + len;
      if (j >= txt.length) { res.push(txt.substring(i).trim()); break; }
      k = txt.lastIndexOf(" ", j);
      if (k <= i) k = j;
      res.push(txt.substring(i, k).trim());
      i = k + 1;
    }
    return res.join("\n");
  }

  function swapTextPreserveLineSlashComments(original, translated) {
    var a = String(original).split(/\r?\n/);
    var b = String(translated).split(/\r?\n/);
    if (a.length !== b.length) return translated;
    var re = /^(\s*)\/\/\s?(.*)$/;
    var out = [];
    for (var i = 0; i < a.length; i++) {
      var m = a[i].match(re);
      if (m) {
        var t = String(b[i] == null ? "" : b[i]).replace(/^\s*\/\/\s?/, "").trim();
        out.push(m[1] + "// " + t);
      } else {
        out.push(b[i]);
      }
    }
    return out.join("\r\n");
  }

      btnSwap = createButton("Swap", function(){
        f.Hide(); clip.SetText(clipboardTextForSwap()); sp.Sleep(20);
        sp.SendModifiedVKeys([vk.LCONTROL], [vk.VK_V]); f.Close();
      });
    }
    if (enablePin) {
      btnPin = createButton("Pin", function(){
        pinHeld.on = !pinHeld.on;
        syncPinButtonText();
        try {
          f.TopMost = true;
          f.BringToFront();
          f.Activate();
        } catch(_) {}
      });
    }
    positionButtons(btnCopy, btnSwap, btnPin, copyPosition);
  } else {
    lbl.Location = new Drawing.Point(f.Padding.All, f.Padding.All);
  }

  var dragging = false, sx = 0, sy = 0;
  function md(s,e){
    if (e.Button !== Forms.MouseButtons.Left) return;
    if (controlIsButton(s, btnCopy) || controlIsButton(s, btnSwap) || controlIsButton(s, btnPin)) return;
    dragging = true;
    sx = e.X;
    sy = e.Y;
  }
  function mm(s,e){ if(dragging){ var p=f.Location; f.Location=new Drawing.Point(p.X+e.X-sx, p.Y+e.Y-sy); } }
  function mu(s,e){ if(e.Button===Forms.MouseButtons.Left) dragging=false; }
  [f, lbl, btnCopy, btnSwap, btnPin].forEach(function(c){
    if (c) {
      c.MouseDown.connect(md); c.MouseMove.connect(mm); c.MouseUp.connect(mu);
      c.DoubleClick.connect(function(){ if (!formIsPinned()) f.Close(); });
    }
  });
  f.KeyPreview = true;
  f.KeyDown.connect(function(s,e){ if(e.KeyCode===Forms.Keys.Escape && !formIsPinned()) f.Close(); });
  f.Deactivate.connect(function(){
    if (formIsPinned()) return;
    if (deactivateCloseTimer) {
      try { deactivateCloseTimer.Stop(); deactivateCloseTimer.Dispose(); } catch(_) {}
      deactivateCloseTimer = null;
    }
    var t = new Forms.Timer();
    t.Interval = 200;
    deactivateCloseTimer = t;
    t.Tick.connect(function(){
      deactivateCloseTimer = null;
      try { t.Stop(); t.Dispose(); } catch(_) {}
      try {
        if (formIsPinned()) return;
        if (!f.Visible) return;
        if (f.ContainsFocus) return;
        if (Forms.Form.ActiveForm === f) return;
        f.Close();
      } catch(_) {}
    });
    t.Start();
  });
  f.Shown.connect(function(){
    f.BringToFront(); f.Activate();
    try { sp.WindowFromHandle(f.Handle).Activate(); } catch(e) {}
    try {
      // Keep popup fully visible inside current screen working area
      var screen = Forms.Screen.FromPoint(sp.GetCurrentMousePoint());
      var wa = screen.WorkingArea;
      var x = f.Location.X, y = f.Location.Y;
      if (x < wa.Left) x = wa.Left;
      if (y < wa.Top) y = wa.Top;
      if (x + f.Width > wa.Right) x = wa.Right - f.Width;
      if (y + f.Height > wa.Bottom) y = wa.Bottom - f.Height;
      if (x < wa.Left) x = wa.Left;
      if (y < wa.Top) y = wa.Top;
      f.Location = new Drawing.Point(x, y);
    } catch(_) {}
  });

  var pt = sp.GetCurrentMousePoint();
  f.StartPosition = Forms.FormStartPosition.Manual;
  f.Location = new Drawing.Point(pt.X + 20, pt.Y + (wrapLen < 30 ? -40 : -150));
  f.ShowDialog();
})();
