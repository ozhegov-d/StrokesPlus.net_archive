  var FF_DEBUG = false;                // log

  var FF_IGNORE_EXECUTABLES = "";  // exe names, case is not important
  var FF_IGNORE_TITLE_CONTAINS = "TextExtractor";  // title
  var FF_IGNORE_CLASSNAME = "WindowsForms10.STATIC.app.0.13965fa_r9_ad1";  // window class name, case is not important

  sp.StoreNumber("FFCurrentIntervalMs", INTERVAL_MS);
  sp.StoreNumber("FFIdleStableTicks", 0);
  sp.StoreBool("FFModalCheckEnabled", FF_MODAL_CHECK_ENABLED);
  sp.StoreNumber("FFModalMaxPercent", FF_MODAL_MAX_PERCENT);
  sp.StoreBool("FFDebug", FF_DEBUG);
  sp.StoreString("FFIgnoreExecutables", FF_IGNORE_EXECUTABLES || "");
  sp.StoreString("FFIgnoreTitleContains", FF_IGNORE_TITLE_CONTAINS || "");
  sp.StoreString("FFIgnoreClassName", FF_IGNORE_CLASSNAME || "");
  sp.StoreBool("FFIgnoreStrict", FF_IGNORE_STRICT);
  sp.StoreString("FFBlockMouseButtons", FF_BLOCK_MOUSE_BUTTONS || "");

    u32tb.DefinePInvokeMethod("GetForegroundWindow", "user32.dll", [], IntPtrT, "PreserveSig");
    u32tb.Create();

      var fgIdleCheckEveryTicks = sp.GetStoredNumber("FFFgIdleCheckEveryTicks") || 3;
      var currentIntervalMs = sp.GetStoredNumber("FFCurrentIntervalMs") || intervalMs;

      if (focusFollowsEnabled === false) return;
      if (now - lastProcessedTick < currentIntervalMs * 0.5) return;
      sp.StoreNumber("FFLastProcessedTick", now);
      if (sp.GetStoredBool("FFBlockActivate") === true) {

          sp.StoreBool("FFBlockActivate", false);
          sp.StoreNumber("FFBlockActivateTime", 0);
        } else { return; }
      }
      var debug = sp.GetStoredBool("FFDebug");
      var listExe = (sp.GetStoredString("FFIgnoreExecutables") || "").toLowerCase();
      var listTitle = (sp.GetStoredString("FFIgnoreTitleContains") || "").toLowerCase();
      var listClass = (sp.GetStoredString("FFIgnoreClassName") || "").toLowerCase();
      var partsExe = listExe ? listExe.split(',') : [];
      var partsTitle = listTitle ? listTitle.split(',') : [];
      var partsClass = listClass ? listClass.split(',') : [];
      var hwndVal = function(h) {

          if (!h) return 0;
          if (typeof h === "number") return h;
          if (typeof h === "string") {
            var parsed = parseInt(h, 10);
            return isNaN(parsed) ? 0 : parsed;
          }
          if (typeof h.ToInt64 === "function") return h.ToInt64();
          if (typeof h.HWnd !== "undefined" && h.HWnd && typeof h.HWnd.ToInt64 === "function") return h.HWnd.ToInt64();
          return 0;
        } catch(e) { sp.ConsoleLog("FF hwndVal ERR: " + e, "FF", System.Drawing.Color.Red); return 0; }
      };
      var FF_BTN_LEFT = 1;
      var FF_BTN_RIGHT = 2;
      var FF_BTN_MIDDLE = 4;
      var FF_BTN_X1 = 8;
      var FF_BTN_X2 = 16;
      var ffBtnNameToBit = { left: FF_BTN_LEFT, right: FF_BTN_RIGHT, middle: FF_BTN_MIDDLE, x1: FF_BTN_X1, x2: FF_BTN_X2 };
      var ffBtnMaskToMouseButton = [
        [FF_BTN_LEFT, MouseButtons.Left],
        [FF_BTN_RIGHT, MouseButtons.Right],
        [FF_BTN_MIDDLE, MouseButtons.Middle],
        [FF_BTN_X1, MouseButtons.XButton1],
        [FF_BTN_X2, MouseButtons.XButton2]
      ];
      var getBlockMouseButtonsMask = function() {
        try {
          var raw = (sp.GetStoredString("FFBlockMouseButtons") || "").toLowerCase();
          var cachedRaw = (sp.GetStoredString("FFBlockMouseButtonsCachedRaw") || "").toLowerCase();

          var mask = 0;
          if (raw) {

            for (var i = 0; i < parts.length; i++) {
              var name = parts[i].trim();
              var bit = ffBtnNameToBit[name];
              if (bit) mask |= bit;
            }
          }
          sp.StoreString("FFBlockMouseButtonsCachedRaw", raw);
          sp.StoreNumber("FFBlockMouseButtonsMask", mask);
          return mask;
        } catch (e) {}
        return 0;

      var shouldBlockByMouseButtons = function() {

          if ((mask & ffBtnMaskToMouseButton[j][0]) !== 0 && sp.IsButtonDown(ffBtnMaskToMouseButton[j][1])) return true;
        }
        return false;
      };
      var ignoredCache = {};
      var isIgnored = function(win) {
        if (!win) return { ignore: false };
        var k = win.HWnd ? win.HWnd.ToString() : "";
        if (k && ignoredCache[k] !== undefined) return ignoredCache[k];

                  return;
                }

            } catch (e) { sp.ConsoleLog("FF catch (modal check): " + e, "FF", System.Drawing.Color.Red); }

        sp.StoreRectangle("FFLastRect", trackRect);
        sp.StoreObject("FocusFollowsTrackedHwnd", root.HWnd);
      } catch (err) { sp.ConsoleLog("FF catch: " + err, "FF", System.Drawing.Color.Red); }
    })();
